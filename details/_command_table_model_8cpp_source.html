MIDI2LR  0.9.9
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

162 {
163  MIDI_Message msg{ midi_channel, midi_data, isCC };
164  if (m_commandMap)

167  {

252 void CommandTableModel::buildFromXml(const XmlElement * const root)
253 {
254  if (root->getTagName().compare("settings") != 0)
255  return;
256 
257  removeAllRows();
258 
259  auto* setting = root->getFirstChildElement();
260  while ((setting) && (m_commandMap))
261  {
262  if (setting->hasAttribute("controller"))
263  {
264  MIDI_Message cc{ setting->getIntAttribute("channel"), setting->getIntAttribute("controller"), true };
265  addRow(cc.channel, cc.controller, true);
266 
267  // older versions of MIDI2LR stored the index of the string, so we should attempt to parse this as well
268  if (setting->getIntAttribute("command", -1) != -1)
269  {
270  m_commandMap->addCommandforMessage(setting->getIntAttribute("command"), cc);
271  }
272  else
273  {
274  m_commandMap->addCommandforMessage(setting->getStringAttribute("command_string"), cc);
275  }
276  }
277  else if (setting->hasAttribute("note"))
278  {
279  MIDI_Message note{ setting->getIntAttribute("channel"), setting->getIntAttribute("note"), false };
280  addRow(note.channel, note.pitch, false);
281 
282  // older versions of MIDI2LR stored the index of the string, so we should attempt to parse this as well
283  if (setting->getIntAttribute("command", -1) != -1)
284  {
285  m_commandMap->addCommandforMessage(setting->getIntAttribute("command"), note);
286  }
287  else
288  {
289  m_commandMap->addCommandforMessage(setting->getStringAttribute("command_string"), note);
290  }
291  }
292  setting = setting->getNextElement();
293  }
294 }
