MIDI2LR  1.0.1.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

33  std::wstring utf8_to_utf16(const std::string& utf8)
34  {
35  std::vector<unsigned long> unicode;
36  size_t i = 0;
37  while (i < utf8.size())
38  {
39  unsigned long uni;
40  size_t todo;
41  bool error = false;
42  unsigned char ch = utf8[i++];
43  if (ch <= 0x7F)
44  {
45  uni = ch;
46  todo = 0;
47  }
48  else if (ch <= 0xBF)
49  {
50  throw std::domain_error("not a UTF-8 string");
51  }
52  else if (ch <= 0xDF)
53  {
54  uni = ch & 0x1F;
55  todo = 1;
56  }
57  else if (ch <= 0xEF)
58  {
59  uni = ch & 0x0F;
60  todo = 2;
61  }
62  else if (ch <= 0xF7)
63  {
64  uni = ch & 0x07;
65  todo = 3;
66  }
67  else
68  {
69  throw std::domain_error("not a UTF-8 string");
70  }
71  for (size_t j = 0; j < todo; ++j)
72  {
73  if (i == utf8.size())
74  throw std::domain_error("not a UTF-8 string");
75  unsigned char ch = utf8[i++];
76  if (ch < 0x80 || ch > 0xBF)
77  throw std::domain_error("not a UTF-8 string");
78  uni <<= 6;
79  uni += ch & 0x3F;
80  }
81  if (uni >= 0xD800 && uni <= 0xDFFF)
82  throw std::domain_error("not a UTF-8 string");
83  if (uni > 0x10FFFF)
84  throw std::domain_error("not a UTF-8 string");
85  unicode.push_back(uni);
86  }
87  std::wstring utf16;
88  for (size_t i = 0; i < unicode.size(); ++i)
89  {
90  unsigned long uni = unicode[i];
91  if (uni <= 0xFFFF)
92  {
93  utf16 += (wchar_t)uni;
94  }
95  else
96  {
97  uni -= 0x10000;
98  utf16 += (wchar_t)((uni >> 10) + 0xD800);
99  utf16 += (wchar_t)((uni & 0x3FF) + 0xDC00);
100  }
101  }
102  return utf16;
103  }
104 }

188 #ifdef _WIN32
189  //Lightroom handle

192  // Bring Lightroom to foreground if it isn't already there

203  }
204  BYTE vk = 0;
205  BYTE vk_modifiers = 0;
206  if (SendKeys::key_map_.count(lower_string))
207  vk = SendKeys::key_map_.at(lower_string);
208  else
209  {// Translate key code to keyboard-dependent scan code, may be UTF-8
210  wchar_t full_character;

212  if (return_value == 0)
213  {

215  if (er == ERROR_INVALID_FLAGS || er == ERROR_INVALID_PARAMETER)
216  throw std::invalid_argument("Bad argument to MultiByteToWideChar.");

218  throw std::length_error("Insufficient buffer for MultiByteToWideChar.");
219  if (er == ERROR_NO_UNICODE_TRANSLATION)
220  throw std::domain_error("Unable to tranlate: MultiByteToWideChar.");
