MIDI2LR  0.9.9
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

49 void CommandMap::addCommandforMessage(unsigned int command, const MIDI_Message &msg)
50 {
51  // adds a msg to the msg:command map, and it's associated command to the command:msg map
52  if (command < LRCommandList::LRStringList.size())
53  {
54  _messageMap[msg] = LRCommandList::LRStringList[command];
55  _commandStringMap[LRCommandList::LRStringList[command]] = msg;
56  }
57  else
58  _messageMap[msg] = LRCommandList::NextPrevProfile[command - LRCommandList::LRStringList.size()];

194 void CommandMap::toXMLDocument(File& file) const
195 {
196  // save the contents of the command map to an xml file
197  XmlElement root{ "settings" };
198  for (auto mapEntry : _messageMap)
199  {
200  auto* setting = new XmlElement{ "setting" };
201  setting->setAttribute("channel", mapEntry.first.channel);
202 
203 
204  setting->setAttribute("NRPN", (mapEntry.first.isNRPN) ? "True" : "False");
205  setting->setAttribute("Relative", (mapEntry.first.isRelative) ? "True" : "False");
206 
207  if (mapEntry.first.isCC)
208  setting->setAttribute("controller", mapEntry.first.controller);
209  else
210  setting->setAttribute("note", mapEntry.first.pitch);
211 
212  setting->setAttribute("command_string", mapEntry.second);
213 
214  root.addChildElement(setting);
215  }
216 
217  if (!root.writeToFile(file, ""))
218  // Give feedback if file-save doesn't work
219  AlertWindow::showMessageBox(AlertWindow::WarningIcon, "File Save Error", "Unable to save file as specified. Please try again, and consider saving to a different location.");
220 }
