MIDI2LR  0.9.9
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

110  auto sizeRead = 0;
111  auto canReadLine = true;
112 
113  // parse input until we have a line, then process that line
114  while (!String(line).endsWithChar('\n') && !threadShouldExit())
115  {
116  auto waitStatus = waitUntilReady(true, 0);
117  if (waitStatus < 0)
118  {
119  canReadLine = false;
120  break;
121  }
122  else if (waitStatus == 0)
123  {
124  wait(100);
125  continue;
126  }
127  sizeRead += read(line + sizeRead, 1, false);
128  }
129 
130  if (canReadLine)
131  {
132  String param{ line };
133  processLine(param);
134  }
135  }

151  line.trimEnd();
152  const auto command = line.upToFirstOccurrenceOf(" ", false, false);
153  const auto valueString = line.replace(line.upToFirstOccurrenceOf(" ", true, true), "", true);
154  const auto value = valueString.getIntValue();
155 
156  if (m_commandMap)
157  {
158 
159  if (command == String{ "SwitchProfile" })
160  {
161  if (m_profileManager)
162  {
163  m_profileManager->switchToProfile(valueString.trim());
164  }
165  }
166  else if (command == String{ "SendKey" })
167  {
168  m_SendKeys.SendKeyDownUp(KeyPress::createFromDescription(valueString.trim()));
169  }
170  else
171  {
172 
173  // store updates in map
174  parameterMap[command] = value;
175 
176  // send associated CC messages to MIDI OUT devices
177  if (m_commandMap->commandHasAssociatedMessage(command))
178  {
179  const auto& msg = m_commandMap->getMessageForCommand(command);
180 
181 
182  if (m_midiSender)
183  {
184  m_midiSender->sendCC(msg.channel, msg.controller, value);
185  }
186  }
187  }
