MIDI2LR  1.0.1.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

48 void CommandMenu::buttonClicked(Button* /*button*/)
49 {
50  size_t index = 1;
51  auto submenu_tick_set = false;
52  PopupMenu main_menu;
53  main_menu.addItem(index, "Unmapped", true, submenu_tick_set = (index == selected_item_));
54  index++;
55 
56  // add each submenu
57  for (size_t menu_index = 0; menu_index < menus_.size(); menu_index++)
58  {
59  PopupMenu subMenu;
60  for (auto command : menu_entries_[menu_index])
61  {
62  auto already_mapped = false;
63  if ((index - 1 < LRCommandList::LRStringList.size()) && (command_map_))
64  {
65  already_mapped = command_map_->commandHasAssociatedMessage(LRCommandList::LRStringList[index - 1]);
66  }
67 
68  // add each submenu entry, ticking the previously selected entry and disabling a previously mapped entry
69 
70  if (already_mapped)
71  subMenu.addColouredItem(index, command, Colours::red, true, index == selected_item_);
72  else
73  subMenu.addItem(index, command, true, index == selected_item_);
74 
75  index++;
76  }
77  // set whether or not the submenu is ticked (true if one of the submenu's entries is selected)
78  main_menu.addSubMenu(menus_[menu_index], subMenu, true, nullptr, selected_item_ < index && !submenu_tick_set);
79  submenu_tick_set |= (selected_item_ < index && !submenu_tick_set);
80  }
81 
82  auto result = static_cast<size_t>(main_menu.show());
83  if ((result) && (command_map_))
84  {
85  // user chose a different command, remove previous command mapping associated to this menu
86  if (selected_item_ < std::numeric_limits<unsigned int>::max())
87  command_map_->removeMessage(message_);
88 
89  if (result - 1 < LRCommandList::LRStringList.size())
90  setButtonText(LRCommandList::LRStringList[result - 1]);
91  else
92  setButtonText(LRCommandList::NextPrevProfile[result - 1 - LRCommandList::LRStringList.size()]);
93 
94  selected_item_ = result;
95 
96  // Map the selected command to the CC
97  command_map_->addCommandforMessage(result - 1, message_);
98  }
99 }
