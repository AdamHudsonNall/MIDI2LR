MIDI2LR  0.9.9
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

359 void LR_IPC_OUT::handleMidiCC(int midiChannel, int controller, int value)
360 {
361  MIDI_Message cc{ midiChannel, controller, true };
362 
363  if (m_commandMap)
364  {
365  if (!m_commandMap->messageExistsInMap(cc) ||
366  m_commandMap->getCommandforMessage(cc) == "Unmapped" ||
367  find(LRCommandList::NextPrevProfile.begin(),
368  LRCommandList::NextPrevProfile.end(),
369  m_commandMap->getCommandforMessage(cc)) != LRCommandList::NextPrevProfile.end())
370  return;
371 
372  _commandToSend = m_commandMap->getCommandforMessage(cc);
373  _valueToSend = value;
374  handleAsyncUpdate();
375  }
376 }
