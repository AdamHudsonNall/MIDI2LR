MIDI2LR  0.9.9
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

361 /**********************************************************************************************//**
362  * @fn void ResizableLayout::recalculateLayout()
363  *
364  * @brief Recalculate the position and size of all the controls in the layout, based on the
365  * owner Component size.
366  *
367  *
368  **************************************************************************************************/
369 
370 void ResizableLayout::recalculateLayout()
371 {
372  if (m_isActive)
373  {
374  Rect rp = m_owner->getBounds();
375 
376  for (int i = 0; i<m_states.size(); i++)
377  {
378  Anchor anchor = m_anchors[i];
379  State state = m_states[i];
380  jassert(anchor.component == state.component);
381 
382  Rect rb;
383 
384  // secret sauce
385  rb.top = state.margin.top + (rp.getHeight() * anchor.topLeft.getY()) / anchorUnit;
386  rb.left = state.margin.left + (rp.getWidth() * anchor.topLeft.getX()) / anchorUnit;
387  rb.bottom = state.margin.bottom + (rp.getHeight() * anchor.bottomRight.getY()) / anchorUnit;
388  rb.right = state.margin.right + (rp.getWidth() * anchor.bottomRight.getX()) / anchorUnit;
389  // prove that there are no rounding issues
390  jassert((anchor.bottomRight != anchor.topLeft) ||
391  ((rb.getWidth() == anchor.component->getWidth()) &&
392  (rb.getHeight() == anchor.component->getHeight())));
393 
394  if (anchor.style == styleStretch)
395  {
396  anchor.component->setBounds(rb);
397  }
398  else if (anchor.style == styleFixedAspect)
399  {
400  Rect r;
401  double aspect = double(rb.getWidth()) / rb.getHeight();
402 
403  if (aspect > state.aspect)
404  {
405  r.top = rb.top;
406  r.bottom = rb.bottom;
407  int width = int(state.aspect * r.getHeight());
408  r.left = rb.left + (rb.getWidth() - width) / 2;
409  r.right = r.left + width;
410  }
411  else
412  {
413  r.left = rb.left;
414  r.right = rb.right;
415  int height = int(1. / state.aspect * r.getWidth());
416  r.top = rb.top + (rb.getHeight() - height) / 2;
417  r.bottom = r.top + height;
418  }
419 
420  anchor.component->setBounds(r);
421  }
422  }
423  }
424 }

504 void ResizableLayout::resizeStart()
505 {
506  bool haveChildren = false;
507 
508  // first recursively call resizeStart() on all children
509  for (int i = 0; i<m_anchors.size(); i++)
510  {
511  ResizableChild* child = m_anchors.getUnchecked(i).child;
512  if (child)
513  {
514  child->resizeStart();
515  haveChildren = true;
516  }
517  }
518 
519  // now check our immediate children for constraints
520  if (haveChildren)
521  {
522  // start with our values
523  int xmin0 = getMinimumWidth();
524  int ymin0 = getMinimumHeight();
525 
526  // for each child, use the inverse of the resize function to solve for the min/max
527  for (int i = 0; i<m_anchors.size(); i++)
528  {
529  Anchor anchor = m_anchors.getUnchecked(i);
530  if (anchor.child)
531  {
532  State state = m_states.getUnchecked(i);
533  jassert(anchor.component == state.component);
534 
535  int xmin1 = anchor.child->getMinimumWidth();
536  int ymin1 = anchor.child->getMinimumHeight();
537  //int xmax1 = anchor.child->getMaximumWidth();
538  //int ymax1 = anchor.child->getMaximumHeight();
539 
540  // invert the sauce
541  int d; // anchor difference
542  int m; // solution to f'(x)
543 
544  d = anchor.bottomRight.getX() - anchor.topLeft.getX();
545  if (d != 0)
546  {
547  m = (xmin1 + state.margin.left - state.margin.right) * anchorUnit / d;
548  xmin0 = jmax(xmin0, m);
549  }
550 
551  d = anchor.bottomRight.getY() - anchor.topLeft.getY();
552  if (d != 0)
553  {
554  m = (ymin1 + state.margin.top - state.margin.bottom) * anchorUnit / d;
555  ymin0 = jmax(ymin0, m);
556  }
557  }
558  }
559 
560  // apply adjusted limits
561  setMinimumWidth(xmin0);
562  setMinimumHeight(ymin0);
563  }
564 }
